# 基础语法

1. 变量定义：

    ```python
        n = 1                               # 定义整型变量
        f = 1.0                             # 定义浮点型变量
        b = True                            # 定义布尔型变量
        s = 'Python'                        # 定义字符串

        a_list = [1, 2, 3]                  # 定义列表
        a_set = {'a', 'b', 'c'}             # 定义集合
        a_dict = {'a': 1, 'b': 2, 'c': 3}   # 定义字典
    ```

    ```c
        int n = 1;                  // 定义整型变量
        double d = 1.0;             // 定义浮点型变量
        int nums[] = {1, 2, 3};     // 定义整型数组
        char s[] = "c";             // 定义字符型数组（字符串）  
    ```

    ```c++
        int n{1};                   // 定义整型变量
        double d{1.0};              // 定义浮点型变量
        bool b{true};               // 定义布尔型变量
        string s{"C++"};            // 定义字符串，<string>模块

        vector<int> aVector{1, 2, 3};                       // 定义列表，<vector>模块
        set<char> aSet = {'a', 'b', 'c'};                   // 定义集合，<set>模块
        map<char, int> aMap{{'a', 1}, {'b', 2}, {'c', 3}};  // 定义字典，<map>模块
    ```

    ```java
        int n = 1;                  // 定义整型变量
        double d = 1.0;             // 定义浮点型变量
        boolean b = true;           // 定义布尔型变量
        String s = "Java";          // 定义字符串

        List<Integer> aList = List.of(1, 2, 3);                         // 定义列表
        Set<Character> aSet = Set.of('a', 'b', 'c');                    // 定义集合
        Map<Character, Integer> aMap = Map.of('a', 1, 'b', 2, 'c', 3);  // 定义映射
    ```
